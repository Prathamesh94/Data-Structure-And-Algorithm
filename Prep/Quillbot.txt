function getLargestSubstring(input){
  let n = input.length
  let map = countCharacter(input,n)
  let maxLength = findLargestSubstring(map,input)
  return maxLength
}

function findLargestSubstring(map,string){
  let low = 0
  let high = string.length-1
  let aCounter = map.get("A")
  let bCounter = map.get("B")
  
  while(low < high){
    //console.log(//)
    if(aCounter == bCounter) return [low,high]
    if(aCounter > bCounter && string[low] == "A"){
      aCounter -= 1
      low += 1
    }else if(aCounter > bCounter && string[high] == "A") {
      aCounter -= 1
      high -=1
    } else if(bCounter > aCounter && string[low] == "B"){
      bCounter -= 1
      low += 1
    } else if(bCounter > aCounter && string[high] == "B"){
      bCounter -= 1
      high -= 1
    }else{
      if(string[low] == "A"){
        aCounter -= 1
      }else{
        bCounter -= 1
      }
      low += 1
    }
  }
  return -1
}

function countCharacter(string,n){
  let map = new Map()

  for(char of string){
    if(map.has(char)){
      map.set(char,map.get(char)+1)
    }else{
      map.set(char,1)
    }
  }
  return map
}
